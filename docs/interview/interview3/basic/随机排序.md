```js
<script>
  var arr =[1,2,3,4];
  var t;
  for(var i = 0;i < arr.length; i++){
    var rand = parseInt(Math.random()*arr.length); 
    console.log(rand,'rand')
    t = arr[rand]; 
    console.log('t',t)
    arr[rand] =arr[i]; 
    console.log(arr[rand],'arr[rand]')
    arr[i] = t;
    console.log('arr[i]',arr[i])
    console.log(arr);
  }
  console.log(arr);
</script>
```

```js
<script>
  var arr =[1,2,3,4];
  var brr = [];
  var num  = arr.length;
  for (var i = 0; i < num; i++){
    var temp = parseInt(Math.random()*(num-i));
    brr.push(arr[temp]);
    arr.splice(temp,1);
  }
  console.log(brr);
</script>
```

```js
<script>
  function randomsort(a, b) {
    return Math.random()>.5 ? -1 : 1; //通过随机产生0到1的数，然后判断是否大于0.5从而影响排序，产生随机性的效果。
  }
  var arr = [1, 2, 3, 4, 5];
  arr.sort(randomsort);
  console.log(arr)
</script>
```
